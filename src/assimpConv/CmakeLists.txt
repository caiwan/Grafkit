# Assimp converter
include (${GK2_ROOT_DIR}/cmake/create_app.cmake)

create_app(AssimpConverter)

# settings
set_target_properties(AssimpConverter PROPERTIES FOLDER tools)
msvc_set_console(AssimpConverter)

if(WIN32)
  target_link_libraries(AssimpConverter wsock32 ws2_32)
endif()

# Additional deps
include_directories("${ASSIMP_INCLUDE_DIR}")
target_link_libraries(AssimpConverter optimized "${ASSIMP_LIBRARY_RELEASE}")
target_link_libraries(AssimpConverter debug "${ASSIMP_LIBRARY_DEBUG}")

add_dependencies(AssimpConverter AssimpConverterPy)

ASSIMP_COPY_BINARIES(AssimpConverter ${CMAKE_CURRENT_BINARY_DIR})

install(TARGETS AssimpConverter DESTINATION tools)
install(FILES "${CMAKE_CURRENT_BINARY_DIR}/${_ASSIMP_DLL_REL}" CONFIGURATIONS Release MinSizeRel DESTINATION tools)
install(FILES "${CMAKE_CURRENT_BINARY_DIR}/${_ASSIMP_DLL_DEB}" CONFIGURATIONS Debug RelWithDebInfo DESTINATION tools)

subdirs(py)
